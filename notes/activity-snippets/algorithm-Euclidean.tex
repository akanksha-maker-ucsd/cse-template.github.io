\begin{algorithm}[caption={Euclidean algorithm for calculating greatest common divisor}]
    procedure $\textit{Euclidean}$($a$: a positive integer, $b$: a positive integer)
    $x$ := $a$
    $y$ := $b$
    while $y \neq 0$
      $r$ := $x \textbf{ mod } y$
      $x$ := $y$
      $y$ := $r$
    return $r$ $\{ r~\textrm{holds the result of } gcd(~(a,b)~)\} $
\end{algorithm}
